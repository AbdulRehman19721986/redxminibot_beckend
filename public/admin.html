Developer Guide
How to write and install custom plugins
Plugin Format
module.exports = [
  {
    pattern: 'hello',
    alias:   ['hi', 'hey'],
    desc:    'Say hello to user',
    category:'Fun',
    execute: async (conn, msg, m, ctx) => {
      ctx.reply('Hello World! 👋');
    }
  }
];
Context (ctx) Object
ctx.from             // chat JID (group or DM)
ctx.sender           // sender JID
ctx.q                // query text after command
ctx.args             // args array
ctx.reply(text)      // send reply
ctx.isOwner          // boolean
ctx.isAdmin          // boolean (group only)
ctx.isGroup          // boolean
ctx.groupMetadata    // group info object
ctx.downloadSessions // Map for number selection
ctx.dlKey            // key for this conversation
m.mentionedJid       // array of @mentioned users
Number Selection Pattern
// Show options, user replies with number
const items = [
  { label: 'Option A', data: '...' },
  { label: 'Option B', data: '...' },
];
ctx.reply('1. Option A\n2. Option B\nReply with number');

ctx.downloadSessions.set(ctx.dlKey, {
  items,
  handler: async (conn, msg2, num, session) => {
    const item = session.items[num - 1];
    ctx.reply('You chose: ' + item.label);
    ctx.downloadSessions.delete(ctx.dlKey);
  }
});
setTimeout(() => ctx.downloadSessions.delete(ctx.dlKey), 120000);
